如何获取两个字符串中的最大公共字符串长度

int getCommonStr(String s1, String s2){ String minStr = s1.length() <= s2.length()?s1:s2; if (minStr.equals(s1)) { for(int i = s1.length(); i > 0; i--) { String temp = s1.substring(0,i); if (s2.contains(temp)) { return i; } } } else if(minStr.equals(s2)){ for(int i = s2.length(); i > 0; i--) { String temp = s2.substring(0,i); if (s1.contains(temp)) { return i; } } } return 0; }
